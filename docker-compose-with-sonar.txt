version: '3.8'

services:
  postgres:
    image: postgres:15-alpine
    container_name: postgres_oficina
    environment:
      POSTGRES_DB: oficina
      POSTGRES_USER: oficina
      POSTGRES_PASSWORD: oficina
    ports:
      - "5432:5432"
    volumes:
      - postgres_data:/var/lib/postgresql/data
    networks:
      - oficina_network
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U oficina"]
      interval: 10s
      timeout: 5s
      retries: 5
  sonar-db:
    image: postgres:15-alpine
    container_name: sonar_db
    environment:
      POSTGRES_DB: sonar
      POSTGRES_USER: sonar
      POSTGRES_PASSWORD: sonar
    volumes:
      - sonar_db_data:/var/lib/postgresql/data
    networks:
      - oficina_network

  sonarqube:
    image: sonarqube:community
    container_name: sonarqube
    depends_on:
      - sonar-db
    ports:
      - "9000:9000"
    environment:
      SONAR_JDBC_URL: jdbc:postgresql://sonar-db:5432/sonar
      SONAR_JDBC_USERNAME: sonar
      SONAR_JDBC_PASSWORD: sonar
    volumes:
      - sonarqube_data:/opt/sonarqube/data
      - sonarqube_logs:/opt/sonarqube/logs
      - sonarqube_extensions:/opt/sonarqube/extensions
    networks:
      - oficina_network

  app:
    build:
      context: .
      dockerfile: Dockerfile
    container_name: minha_app_spring
    depends_on:
      postgres:
        condition: service_healthy
    ports:
      - "8080:8080"
    environment:
      SPRING_DATASOURCE_URL: jdbc:postgresql://postgres:5432/oficina
      SPRING_DATASOURCE_USERNAME: oficina
      SPRING_DATASOURCE_PASSWORD: oficina
    networks:
      - oficina_network

volumes:
  postgres_data:
  sonar_db_data:
  sonarqube_data:
  sonarqube_logs:
  sonarqube_extensions:

networks:
  oficina_network:
